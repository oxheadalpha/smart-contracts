$schema:
	{ "add_implicit_owners": [ $key_hash , ... ] } || 
	{ "burn_tokens": $burn_tokens } || 
	{ "create_token": $create_token } || 
	{ "mint_tokens": $mint_tokens } || 
	{ "pause": bool  /* pause */ } || 
	{ "remove_implicit_owners": [ $key_hash , ... ] } || 
	{ "set_admin": $address  /* set_admin */ } || 
	{ "add_operator": $address  /* add_operator */ } || 
	{ "balance_of": $balance_of } || 
	{ "is_operator": $is_operator } || 
	{ "remove_operator": $address  /* remove_operator */ } || 
	{ "transfer": $transfer }

$transfer:
	{
	  "batch": [ $batch_item , ... ],'
	  "data": $bytes,
	  "from_": $address, 
	  "to_": $address
	}

$batch_item:
	{
	  "amount": $nat,
	  "token_id": $nat
	}

$is_operator:
	{
	  "is_operator_request": $is_operator_request,
	  "is_operator_view": $contract (pair (pair (address %operator) (address %owner)) bool)
	}

$is_operator_request:
	{
	  "operator": $address,
	  "owner": $address
	}

$balance_of:
	{
	  "balance_request": [ $balance_request_item , ... ],
	  "balance_view": $contract (list (pair (pair (address %owner) (nat %token_id)) nat))
	}

$balance_request_item:
	{
	  "owner": $address,
	  "token_id": $nat
	}

$mint_tokens:
	{
	  "batch": [ $batch_item , ... ],
	  "data": $bytes,
	  "owner": $address
	}

$batch_item:
	{
	  "amount": $nat,
	  "token_id": $nat
	}

$create_token:
	{
	  "descriptor": string,
	  "token_id": $nat
	}

$burn_tokens:
	{
	  "batch": [ $batch_item , ... ],
	  "owner": $address
	}

$batch_item:
	{
	  "amount": $nat,
	  "token_id": $nat
	}

$nat:
	int  /* Natural number */

$key_hash:
	string  /* Base58 encoded public key hash */

$contract:
	string  /* Base58 encoded `KT` address */

$bytes:
	string  /* Hex string */ ||
	bytes  /* Python byte string */

$address:
	string  /* Base58 encoded `tz` or `KT` address */
